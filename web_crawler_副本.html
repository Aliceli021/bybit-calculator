import requests

from bs4 import BeautifulSoup

import csv

from datetime import datetime


class WebCrawler:

    def __init__(self, url):

        self.url = url

        self.headers = {

            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'

        }


    def fetch_page(self):

        try:

            response = requests.get(self.url, headers=self.headers)

            response.raise_for_status()

            return response.text

        except requests.RequestException as e:

            print(f"获取页面时发生错误: {e}")

            return None


    def parse_content(self, html):

        if not html:

            return []

        

        soup = BeautifulSoup(html, 'html.parser')

        # 这里以提取所有段落为例

        paragraphs = soup.find_all('p')

        return [p.text.strip() for p in paragraphs]


    def save_to_file(self, data):

        filename = f"crawled_data_{datetime.now().strftime('%Y%m%d_%H%M%S')}.csv"

        try:

            with open(filename, 'w', newline='', encoding='utf-8') as f:

                writer = csv.writer(f)

                writer.writerow(['Content'])

                for item in data:

                    writer.writerow([item])

            print(f"数据已保存到 {filename}")

        except IOError as e:

            print(f"保存文件时发生错误: {e}")


def main():

    # 示例使用

    url = "https://example.com"  # 替换为你想爬取的网站

    crawler = WebCrawler(url)

    

    # 获取页面内容

    html_content = crawler.fetch_page()

    

    # 解析内容

    parsed_data = crawler.parse_content(html_content)

    

    # 保存数据

    crawler.save_to_file(parsed_data)


if __name__ == "__main__":

    main()
